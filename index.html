        // De data structuur

        const locations = {

            2: { title: "Modellen (Model)", type: "long", route: true, next: 3,

            3: { title: "Camp Street", type: "short", route: true, next: 4, 

                 text: "Dit is de Lagerstraße. Hier werden gevangenen binnengebracht. Het scheidt de wereld van de daders (SS) van de wereld van de slachtoffers.",

                 walking: "Loop door naar het SS-troepenkwartier aan de rechterkant." },

            // Overige locaties (placeholders voor nu)

            4: { title: "SS Troop Camp", type: "short", route: true, next: 6, text: "Informatie over het SS-kamp...", sources: ["Bronvermelding volgt"], walking: "Loop naar het administratieve hoofdkwartier." },

            6: { title: "Command Headquarters", type: "short", route: true, next: 7, text: "Informatie over het hoofdkwartier...", sources: ["Bronvermelding volgt"], walking: "Loop naar de woning van de commandant." },

            // ... (Hier vullen we alle 40 nummers aan)

        };


        // Functie om alle 40 nummers te genereren voor het menu

        function generateFullMenu() {

            let html = `<h2 class="text-xl font-bold mb-4">Alle Locaties</h2><div class="grid grid-cols-4 gap-2">`;

            for (let i = 1; i <= 40; i++) {

                const isRoute = locations[i] && locations[i].route;

                html += `<button onclick="loadLocation(${i})" class="${isRoute ? 'bg-blue-600 text-white' : 'bg-white text-gray-600'} border p-3 rounded shadow-sm font-bold">${i}</button>`;

            }

            html += `</div><p class="mt-4 text-sm text-gray-500">* Blauw = Onderdeel van de route</p>`;

            return html;

        }


        function showMenu() {

            document.getElementById('app-content').innerHTML = generateFullMenu();

        }


        function loadLocation(id) {

            const loc = locations[id];

            let html = "";

            

            if (!loc) {

                html = `<div class="bg-white p-6 rounded-lg shadow">

                            <h2 class="text-2xl font-bold mb-2">Locatie ${id}</h2>

                            <p class="text-gray-700">Beknopte informatie over dit punt (nog in te vullen)...</p>

                            <button onclick="showMenu()" class="mt-6 w-full bg-slate-800 text-white font-bold py-3 rounded">Terug naar menu</button>

                        </div>`;

            } else {

                html = `<div class="bg-white p-6 rounded-lg shadow">

                            <div class="flex justify-between items-start mb-4">

                                <span class="bg-slate-200 text-slate-800 px-2 py-1 rounded text-xs font-bold uppercase">${loc.type === 'long' ? 'Lange stop' : 'Korte stop'}</span>

                                <span class="text-slate-400 font-bold text-xl">#${id}</span>

                            </div>

                            <h2 class="text-2xl font-bold mb-4">${loc.title}</h2>

                            <div class="prose text-gray-800 leading-relaxed space-y-4">

                                <p>${loc.text}</p>

                                ${loc.extra ? `<div class="bg-amber-50 border-l-4 border-amber-400 p-3 italic font-medium">${loc.extra}</div>` : ''}

                            </div>

                            

                            <div class="mt-8 pt-6 border-t border-gray-100">

                                <button onclick="finishLocation(${id})" class="w-full bg-green-600 hover:bg-green-700 text-white font-bold py-4 rounded-xl shadow-lg transition-colors">

                                    IK BEN KLAAR

                                </button>

                            </div>


                            <div id="next-step-${id}" class="hidden mt-6 p-4 bg-blue-50 border border-blue-200 rounded-lg">

                                <p class="text-sm font-bold text-blue-800 mb-1">Volgende stap:</p>

                                <p class="text-blue-900">${loc.walking}</p>

                                <button onclick="loadLocation(${loc.next})" class="mt-3 w-full bg-blue-600 text-white py-2 rounded font-bold">Ga naar #${loc.next}</button>

                            </div>


                            <div class="mt-10">

                                <h3 class="text-xs font-bold text-gray-400 uppercase tracking-widest mb-2">Bronnen voor de docent</h3>

                                <ul class="text-sm text-gray-500 list-disc ml-4">

                                    ${loc.sources.map(s => `<li>${s}</li>`).join('')}

                                </ul>

                            </div>

                        </div>`;

            }

            

            document.getElementById('app-content').innerHTML = html;

            window.scrollTo(0,0);

        }


        function finishLocation(id) {

            const nextDiv = document.getElementById(`next-step-${id}`);

            if (nextDiv) {

                nextDiv.classList.remove('hidden');

                nextDiv.scrollIntoView({ behavior: 'smooth' });

            } else {

                showMenu();

            }

        }


        // Start de app op het menu

        showMenu();
